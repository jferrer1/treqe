(function() {
  var steps = document.querySelectorAll('.step');
  if (!steps.length) return;
  if ('IntersectionObserver' in window) {
    var observer = new IntersectionObserver(function(entries) {
      entries.forEach(function(entry) {
        if (entry.isIntersecting) {
          entry.target.classList.add('visible');
        }
      });
    }, { threshold: 0.15 });
    steps.forEach(function(s, i) {
      s.style.transitionDelay = (i * 0.1) + 's';
      observer.observe(s);
    });
  } else {
    steps.forEach(function(s) { s.classList.add('visible'); });
  }
})();

function toggleSearch() {
  var icon = document.getElementById('searchIcon');
  var expand = document.getElementById('searchExpand');
  var input = document.getElementById('searchInput');
  if (expand.classList.contains('open')) {
    expand.classList.remove('open');
    setTimeout(function() { icon.style.opacity = '1'; }, 200);
    if (input.value.trim()) {
      window.location.href = '../v1-catalogo/?search=' + encodeURIComponent(input.value.trim());
    }
  } else {
    icon.style.opacity = '0';
    expand.classList.add('open');
    setTimeout(function() { input.focus(); }, 300);
  }
}
document.addEventListener('DOMContentLoaded', function() {
  var si = document.getElementById('searchInput');
  if (si) si.addEventListener('keydown', function(e) {
    if (e.key === 'Enter' && this.value.trim()) {
      window.location.href = '../v1-catalogo/?search=' + encodeURIComponent(this.value.trim());
    }
  });
});


(function() {
  if (localStorage.getItem('treqe-darkmode') === 'true') {
    document.body.classList.add('dark');
  }
})();

</script>
